add_executable(CubeVoid main.cpp)

target_link_libraries(CubeVoid 
    Core 
    Render 
    Simulation
    OpenGL::GL
    ${CONAN_LIBS}
)

                   COMMAND ${CMAKE_COMMAND} -E copy_directory
                   ${PROJECT_SOURCE_DIR}/src/games/cubevoid/install $<TARGET_FILE_DIR:CubeVoid>)
